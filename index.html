export function loadState(){
  try{
    const raw = localStorage.getItem(KEY);
    return raw ? JSON.parse(raw) : {};
  }catch{
    return {};
  }
}

export function saveState(patch){
  const cur = loadState();
  const next = { ...cur, ...patch };
  localStorage.setItem(KEY, JSON.stringify(next));
  return next;
}
